67 /// <summary>Find a method of a type by its name.</summary>
68 public static MethodInfo FindMethod(this Type type, string methodName, Type[] args = null, bool throwNotFound = true) {
69 if (type == null)
70 throw new ArgumentNullException("type");
71
72 MethodInfo method;
73
74 if (args == null) {
75 method = type.GetMethod(methodName, FULL_BINDING);
76 // method = type.GetMethods(FULL_BINDING)
77 // .Where(m => m.Name == methodName)
78 // .FirstOrDefault();
79 } else {
80 method = type.GetMethod(methodName, FULL_BINDING, null, args, null);
81
82 // There are very specific cases where the above method may not bind properly
83 // e.g. when the method declares an enum and the arg type is an int, so we ignore the args
84 // and hope that there are no ambiguity of methods
85 if (method == null) {
86 method = FindMethod(type, methodName, null, throwNotFound);
87
88 if (method != null && method.GetParameters().Length != args.Length)
89 method = null;
90 }
91 }
92
93 if (method == null && throwNotFound)
94 throw new MissingMethodException(type.FullName, methodName);
95
96 return method;
97 }
