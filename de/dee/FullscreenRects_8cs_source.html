34 /// <summary>Returns a fullscreen rect</summary>
35 /// <param name="mode">The mode that will be used to retrieve the rect.</param>
36 /// <param name="targetWindow">The window that will be set fullscreen.</param>
37 public static Rect GetFullscreenRect(RectSourceMode mode, ScriptableObject targetWindow = null) {
38
39 if (targetWindow != null && !targetWindow.IsOfType(typeof(EditorWindow)) && !targetWindow.IsOfType(Types.View)) {
40 throw new ArgumentException("Target window must be of type EditorWindow or View or null", "targetWindow");
41 }
42
43 if (CustomRectCallback != null) {
44 var rect = new Rect();
45 var shouldUse = CustomRectCallback(mode, out rect);
46
47 if (shouldUse)
48 return rect;
49 }
50
51 switch (mode) {
52 case RectSourceMode.MainDisplay:
53 return GetMainDisplayRect();
54
55 case RectSourceMode.WindowDisplay:
56 if (targetWindow == null || !FullscreenUtility.IsWindows)
57 return GetMainDisplayRect();
58
59 var views = new ViewPyramid(targetWindow);
60 var rect = views.Container.GetPropertyValue<Rect>("position");
61
62 return GetDisplayBoundsAtPoint(rect.center);
63
64 case RectSourceMode.AtMousePosition:
65 return FullscreenUtility.IsWindows ?
66 GetDisplayBoundsAtPoint(FullscreenUtility.MousePosition) :
67 GetWorkAreaRect(true);
68
69 case RectSourceMode.Span:
70 return FullscreenUtility.IsWindows ?
71 GetVirtualScreenBounds() :
72 GetWorkAreaRect(true);
73
74 case RectSourceMode.Custom:
75 return GetCustomUserRect();
76
77 case RectSourceMode.Display1:
78 return GetMonitorRect(0);
79 case RectSourceMode.Display2:
80 return GetMonitorRect(1);
81 case RectSourceMode.Display3:
82 return GetMonitorRect(2);
83 case RectSourceMode.Display4:
84 return GetMonitorRect(3);
85 case RectSourceMode.Display5:
86 return GetMonitorRect(4);
87 case RectSourceMode.Display6:
88 return GetMonitorRect(5);
89 case RectSourceMode.Display7:
90 return GetMonitorRect(6);
91 case RectSourceMode.Display8:
92 return GetMonitorRect(7);
93
94 default:
95 Logger.Warning("Invalid fullscreen mode, please fix this by changing the placement source mode in preferences.");
96 return new Rect(Vector2.zero, Vector2.one * 300f);
97 }
98 }

100 /// <summary>Returns a rect with the dimensions of the main screen.
101 /// (Note that the position may not be right for multiple screen setups)</summary>
102 public static Rect GetMainDisplayRect() {
103
104 if (FullscreenUtility.IsWindows) {
105 var mainDisplay = DisplayInfo
106 .GetDisplays()
107 .FirstOrDefault(d => d.PrimaryDisplay);
108
109 if (mainDisplay != null)
110 return mainDisplay.UnityCorrectedArea;
111
112 Logger.Error("No main display??? This should not happen, falling back to Screen.currentResolution");
113 }
114
115 // Screen.currentResolution returns the resolution of the screen where
116 // the currently focused window is located, not the main display resolution.
117 // This caused the bug #53 on windows.
118 // The same behaviour was not tested on Linux as macOS
119 return new Rect(0f, 0f, Screen.currentResolution.width, Screen.currentResolution.height);
120 }

122 /// <summary>Returns the rect of a given display index.</summary>

143 /// <summary>Returns a rect covering all the screen, except for the taskbar/dock.
144 /// On Windows it adds a 4px border and does not account for scaling (can cause bugs when using scales different than 100%).

166 /// <summary>Full virtual screen bounds, spanning across all monitors. (Windows only)</summary>
167 public static Rect GetVirtualScreenBounds() {
168
169 if (!FullscreenUtility.IsWindows)
170 throw new NotImplementedException();
171
172 const int SM_XVIRTUALSCREEN = 76;
173 const int SM_YVIRTUALSCREEN = 77;
174 const int SM_CXVIRTUALSCREEN = 78;
175 const int SM_CYVIRTUALSCREEN = 79;
176
177 var x = User32.GetSystemMetrics(SM_XVIRTUALSCREEN);
178 var y = User32.GetSystemMetrics(SM_YVIRTUALSCREEN);
179 var width = User32.GetSystemMetrics(SM_CXVIRTUALSCREEN);
180 var height = User32.GetSystemMetrics(SM_CYVIRTUALSCREEN);
181
182 var rect = new Rect {
183 yMin = y,
184 xMin = x,
185 width = width,
186 height = height,
187 };
188
189 return FullscreenUtility.DpiCorrectedArea(rect);
