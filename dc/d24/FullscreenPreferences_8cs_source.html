188 private static void OnPreferencesGUI(string search) {
189
190 ToolbarVisible.DoGUI();
191 FullscreenOnPlayEnabled.DoGUI();
192
193 EditorGUILayout.Separator();
194 RectSource.DoGUI();
195
196 if (RectSource.Value == RectSourceMode.AtMousePosition)
197 EditorGUILayout.HelpBox("\'At mouse position\' can cause slowdowns on large projects", MessageType.Warning);
198
199 if (!IsRectModeSupported(RectSource))
200 EditorGUILayout.HelpBox("The selected placement source mode is not supported on this platform", MessageType.Warning);
201
202 // Custom Rect
203 switch (RectSource.Value) {
204 case RectSourceMode.Custom:
205 EditorGUI.indentLevel++;
206 CustomRect.DoGUI();
207
208 var customRect = CustomRect.Value;
209
210 if (customRect.width < 300f)
211 customRect.width = 300f;
212 if (customRect.height < 300f)
213 customRect.height = 300f;
214
215 CustomRect.Value = customRect;
216
217 EditorGUI.indentLevel--;
218 break;
219 }
220
221 DisableNotifications.DoGUI();
222 KeepFullscreenBelow.DoGUI();
223
224 if (Patcher.IsSupported())
225 DisableSceneViewRendering.DoGUI();
226
227 UseGlobalToolbarHiding.DoGUI();
228 RestoreCursorLockAndHideState.DoGUI();
229
230 if (FullscreenUtility.IsLinux) {
231 using (new EditorGUI.DisabledGroupScope(!FullscreenEditor.Linux.wmctrl.IsInstalled)) {
232 DoNotUseWmctrl.DoGUI();
233 }
234 if (!FullscreenEditor.Linux.wmctrl.IsInstalled) {
235 EditorGUILayout.HelpBox("'wmctrl' not found. Try installing it with 'sudo apt-get install wmctrl'.", MessageType.Warning);
236 } else {
237 EditorGUILayout.HelpBox("Try enabling the option above if you're experiencing any kind of toolbars or offsets " +
238 "while in fullscreen mode.\nDisabling 'wmctrl' can fix issues on some Linux environments when the window manager " +
239 "does not handle fullscreen windows properly (I'm looking at you Ubuntu).", MessageType.Info);
240 }
241 }
242
243 // Mosaic
244 if (FullscreenRects.ScreenCount > 1) {
245 EditorGUILayout.Separator();
246 EditorGUILayout.LabelField(MosaicMapping.Content, EditorStyles.boldLabel);
247 EditorGUI.indentLevel++;
248 GUI.changed = false;
249 var mosaicMapping = MosaicMapping.Value;
250
251 for (var i = 0; i < mosaicMapping.Length && i < FullscreenRects.ScreenCount; i++) {
252 var val = EditorGUILayout.IntPopup(string.Format("Physical display {0} renders", i + 1), mosaicMapping[i], mosaicDropDownOptions, new[] { -1, 0, 1, 2, 3, 4, 5, 6, 7 });
253 mosaicMapping[i] = val;
254 }
255
256 if (GUI.changed)
257 MosaicMapping.SaveValue();
258 EditorGUI.indentLevel--;
259 }
260 }

379 internal static bool IsRectModeSupported(RectSourceMode mode) {
380 switch (mode) {
381 case RectSourceMode.Display1:
382 case RectSourceMode.Display2:
383 case RectSourceMode.Display3:
384 case RectSourceMode.Display4:
385 case RectSourceMode.Display5:
386 case RectSourceMode.Display6:
387 case RectSourceMode.Display7:
388 case RectSourceMode.Display8:
389 case RectSourceMode.Span:
390 case RectSourceMode.WindowDisplay:
391 return FullscreenUtility.IsWindows;
392
393 case RectSourceMode.MainDisplay:
394 case RectSourceMode.AtMousePosition:
395 return true;
396
397 case RectSourceMode.Custom:
398 // Custom rect is not supported on Linux
399 // since we're using native fullscreen
400 return !FullscreenUtility.IsLinux;
401
402 default:
403 return false;
404 }
405 }
