49 private void LoadValue() {
50 try {
51 if (EditorPrefs.HasKey(Key))
52 JsonUtility.FromJsonOverwrite(EditorPrefs.GetString(Key), this);
53 } catch (Exception e) {
54 Logger.Warning("Failed to load {0}, using default value: {1}", Key, e);
55 savedValue = DefaultValue;
56 SaveValue();
57 }
58 }
59
60 public void SaveValue() {
61 try {
62 EditorPrefs.SetString(Key, JsonUtility.ToJson(this));
63 Logger.Debug("Saved value to key {0}:\n{1}", Key, EditorPrefs.GetString(Key));
64
65 if (OnValueSaved != null)
66 OnValueSaved.Invoke(Value);
67
68 InternalEditorUtility.RepaintAllViews();
69 } catch (Exception e) {
70 Logger.Warning("Failed to save {0}: {1}", Key, e);
71 }
72 }
73
74 public void DeleteValue() {
75 EditorPrefs.DeleteKey(Key);
76 savedValue = DefaultValue;
77
78 if (OnValueSaved != null)
79 OnValueSaved.Invoke(savedValue);
80 }
