139 private void GetRenderDirections(Vector3 headPosition, out int eastWestMask, out int northSouthMask) {
140 const int leftMask = 1 << (int)CubemapFace.PositiveX;
141 const int rightMask = 1 << (int)CubemapFace.NegativeX;
142 const int topMask = 1 << (int)CubemapFace.PositiveY;
143 const int bottomMask = 1 << (int)CubemapFace.NegativeY;
144 const int frontMask = 1 << (int)CubemapFace.PositiveZ;
145 const int backMask = 1 << (int)CubemapFace.NegativeZ;
146
147 eastWestMask = 0;
148 northSouthMask = 0;
149 Vector3 southWestBoundary = Vector3.zero;
150 Vector3 northEastBoundary = Vector3.zero;
151 Vector3 northWestBoundary = Vector3.zero;
152 Vector3 southEastBoundary = Vector3.zero;
153
154 // Get North-East and South-West boundaries where the sampled cubemap switches for stereoscopic rendering.
155 List<float> xIntersectSouthWestToNorthEast = MathsUtil.SolveQuadraticEquation(
156 1.0f,
157 headPosition.x + headPosition.z,
158 -0.5f * (cavernRadius * cavernRadius - headPosition.x * headPosition.x - headPosition.z * headPosition.z));
159 // Get North-West and South-East boundaries where the sampled cubemap switches for stereoscopic rendering.
160 List<float> xIntersectNorthWestToSouthEast = MathsUtil.SolveQuadraticEquation(
161 1.0f,
162 headPosition.x - headPosition.z,
163 -0.5f * (cavernRadius * cavernRadius - headPosition.x * headPosition.x - headPosition.z * headPosition.z));
164 if (xIntersectSouthWestToNorthEast.Count == 1) {
165 northEastBoundary = new Vector3(xIntersectSouthWestToNorthEast[0], 0.0f, xIntersectSouthWestToNorthEast[0]);
166 southWestBoundary = new Vector3(xIntersectSouthWestToNorthEast[0], 0.0f, xIntersectSouthWestToNorthEast[0]);
167 } else if (xIntersectSouthWestToNorthEast.Count == 2) {
168 northEastBoundary = new Vector3(xIntersectSouthWestToNorthEast[1], 0.0f, xIntersectSouthWestToNorthEast[1]);
169 southWestBoundary = new Vector3(xIntersectSouthWestToNorthEast[0], 0.0f, xIntersectSouthWestToNorthEast[0]);
170 }
171
172 if (xIntersectNorthWestToSouthEast.Count == 1) {
173 northWestBoundary = new Vector3(xIntersectNorthWestToSouthEast[0], 0.0f, -xIntersectNorthWestToSouthEast[0]);
174 southEastBoundary = new Vector3(xIntersectNorthWestToSouthEast[0], 0.0f, -xIntersectNorthWestToSouthEast[0]);
175 } else if (xIntersectNorthWestToSouthEast.Count == 2) {
176 northWestBoundary = new Vector3(xIntersectNorthWestToSouthEast[0], 0.0f, -xIntersectNorthWestToSouthEast[0]);
177 southEastBoundary = new Vector3(xIntersectNorthWestToSouthEast[1], 0.0f, -xIntersectNorthWestToSouthEast[1]);
178 }
179
180 // Edge Case 1: Head is moved out of the screen area and there are no intersects.
181 // This means that the screen is entirely in one quadrant relative to the head.
182 if (xIntersectSouthWestToNorthEast.Count == 0 && xIntersectNorthWestToSouthEast.Count == 0) {
183 // Screen is south of the head.
184 if (0.0f < head.localPosition.z &&
185 Mathf.Abs(head.localPosition.x) < Mathf.Abs(head.localPosition.z)) {
186 // For edge cases, fuck it just render the top and the bottom too.
187 // I can't be bothered to check those as well, since this should rarely ever happen.
188 eastWestMask |= (backMask | topMask | bottomMask);
189 LogDirection("[East-West] Back, Top, Bottom");
190 return;
191 }
192
193 // Screen is north of the head.
194 if (head.localPosition.z < 0.0f &&
195 Mathf.Abs(head.localPosition.x) < Mathf.Abs(head.localPosition.z)) {
196 eastWestMask |= (frontMask | topMask | bottomMask);
197 LogDirection("[East-West] Front, Top, Bottom");
198 return;
199 }
200
201 // Screen is east of the head.
202 if (head.localPosition.x < 0.0f &&
203 Mathf.Abs(head.localPosition.z) < Mathf.Abs(head.localPosition.x)) {
204 northSouthMask |= (rightMask | topMask | bottomMask);
205 LogDirection("[North-South] Right, Top, Bottom");
206 return;
207 }
208
209 // Screen is west of the head.
210 if (head.localPosition.x > 0.0f &&
211 Mathf.Abs(head.localPosition.z) < Mathf.Abs(head.localPosition.x)) {
212 northSouthMask |= (leftMask | topMask | bottomMask);
213 LogDirection("[North-South] Left, Top, Bottom");
214 return;
215 }
216 }
217
218 // Edge Case 2: Head is moved out of the screen and only the South-West to North-East line intersects.
219 if (xIntersectSouthWestToNorthEast.Count > 0 && xIntersectNorthWestToSouthEast.Count == 0) {
220 // Screen is south-west of the head.
221 if (Vector3.Dot(new Vector3(1.0f, 1.0f), new Vector2(head.localPosition.x, head.localPosition.z)) > 1.0f) {
222 eastWestMask |= (backMask | topMask | bottomMask);
223 northSouthMask |= (leftMask | topMask | bottomMask);
224 LogDirection("[East-West] Back, Top, Bottom\n[North-South] Left, Top, Bottom");
225 return;
226 }
227
228 // Screen is north-east of the head.
229 if (Vector3.Dot(new Vector3(-1.0f, -1.0f), new Vector2(head.localPosition.x, head.localPosition.z)) > 1.0f) {
230 eastWestMask |= (frontMask | topMask | bottomMask);
231 northSouthMask |= (rightMask | topMask | bottomMask);
232 LogDirection("[East-West] Front, Top, Bottom\n[North-South] Right, Top, Bottom");
233 return;
234 }
235 }
236
237 // Edge Case 3: Head is moved out of the screen and only the North-West to South-East line intersects.
238 if (xIntersectSouthWestToNorthEast.Count == 0 && xIntersectNorthWestToSouthEast.Count > 0) {
239 // Screen is north-west of the head.
240 if (Vector3.Dot(new Vector3(1.0f, -1.0f), new Vector2(head.localPosition.x, head.localPosition.z)) > 1.0f) {
241 eastWestMask |= (frontMask | topMask | bottomMask);
242 northSouthMask |= (leftMask | topMask | bottomMask);
243 LogDirection("[East-West] Front, Top, Bottom\n[North-South] Left, Top, Bottom");
244 return;
245 }
246
247 // Screen is south-east of the head.
248 if (Vector3.Dot(new Vector3(-1.0f, 1.0f), new Vector2(head.localPosition.x, head.localPosition.z)) > 1.0f) {
249 eastWestMask |= (backMask | topMask | bottomMask);
250 northSouthMask |= (rightMask | topMask | bottomMask);
251 LogDirection("[East-West] Back, Top, Bottom\n[North-South] Right, Top, Bottom");
252 return;
253 }
254 }
255
256 // Regular Case: The head is within the screen area.
257 string eastWestDebugMsg = "[East-West] Front";
258 string northSouthDebugMsg = "\n[North-South]";
259 eastWestMask |= frontMask; // Always render the front.
260 northSouthMask = 0;
261
262 float screenTop = cavernElevation + cavernHeight - headPosition.y;
263 float screenBottom = cavernElevation - headPosition.y;
264 Vector3 headOffset = new Vector3(headPosition.x, 0.0f, headPosition.z);
265
266 // Calculate the render mask for the East and West cubemaps.
267 if (Vector3.Angle(headOffset + southWestBoundary, Vector3.forward) < cavernAngle * 0.5f || Vector3.Angle(headOffset + southEastBoundary, Vector3.forward) < cavernAngle * 0.5f) {
268 eastWestMask |= backMask;
269 eastWestDebugMsg += ", Back";
270 }
271 if (Mathf.Abs(northEastBoundary.z) < Mathf.Abs(screenTop) ||
272 Mathf.Abs(northWestBoundary.z) < Mathf.Abs(screenTop) ||
273 Mathf.Abs(southEastBoundary.z) < Mathf.Abs(screenTop) ||
274 Mathf.Abs(southWestBoundary.z) < Mathf.Abs(screenTop)) {
275 eastWestMask |= topMask;
276 eastWestDebugMsg += ", Top";
277 }
278 if (Mathf.Abs(northEastBoundary.z) < Mathf.Abs(screenBottom) ||
279 Mathf.Abs(northWestBoundary.z) < Mathf.Abs(screenBottom) ||
280 Mathf.Abs(southEastBoundary.z) < Mathf.Abs(screenBottom) ||
281 Mathf.Abs(southWestBoundary.z) < Mathf.Abs(screenBottom)) {
282 eastWestMask |= bottomMask;
283 eastWestDebugMsg += ", Bottom";
284 }
285
286 // Calculate the render mask for the North and South cubemaps.
287 if (Vector3.Angle(headOffset + northEastBoundary, Vector3.forward) < cavernAngle * 0.5f || Vector3.Angle(headOffset + southEastBoundary, Vector3.forward) < cavernAngle * 0.5f) {
288 northSouthMask |= rightMask;
289 northSouthDebugMsg += " Right";
290 }
291 if (Vector3.Angle(headOffset + northWestBoundary, Vector3.forward) < cavernAngle * 0.5f || Vector3.Angle(headOffset + southWestBoundary, Vector3.forward) < cavernAngle * 0.5f) {
292 northSouthMask |= leftMask;
293 northSouthDebugMsg += ", Left";
294 }
295 if (Mathf.Abs(northEastBoundary.x) < Mathf.Abs(screenTop) ||
296 Mathf.Abs(northWestBoundary.x) < Mathf.Abs(screenTop) ||
297 Mathf.Abs(southEastBoundary.x) < Mathf.Abs(screenTop) ||
298 Mathf.Abs(southWestBoundary.x) < Mathf.Abs(screenTop)) {
299 northSouthMask |= topMask;
300 northSouthDebugMsg += ", Top";
301 }
302 if (Mathf.Abs(northEastBoundary.x) < Mathf.Abs(screenBottom) ||
303 Mathf.Abs(northWestBoundary.x) < Mathf.Abs(screenBottom) ||
304 Mathf.Abs(southEastBoundary.x) < Mathf.Abs(screenBottom) ||
305 Mathf.Abs(southWestBoundary.x) < Mathf.Abs(screenBottom)) {
306 northSouthMask |= bottomMask;
307 northSouthDebugMsg += ", Bottom";
308 }
309
310 LogDirection(eastWestDebugMsg + northSouthDebugMsg);
311 }

313 private void RenderEyes() {
314 // If clampHeadPosition is true, limit the head position to be within the bounds of the circle.
315 if (clampHeadPosition) {
316 Vector2 horizontalPosition = new Vector2(head.transform.localPosition.x, head.transform.localPosition.z);
317 if (horizontalPosition.sqrMagnitude > clampHeadRatio * clampHeadRatio * cavernRadius * cavernRadius) {
318 horizontalPosition = horizontalPosition.normalized * clampHeadRatio * cavernRadius;
319 head.transform.localPosition = new Vector3(horizontalPosition.x, head.transform.localPosition.y, horizontalPosition.y);
320 }
321 }
322
323 // Use Camera.MonoOrStereoscopicEye.Left or Camera.MonoOrStereoscopicEye.Right to ensure that the cubemap follows the camera's rotation.
324 // Camera.MonoOrStereoscopicEye.Mono renders the cubemap to be aligned to the world's axes instead.
325 int eastWestMask = 0;
326 int northSouthMask = 0;
327 GetRenderDirections(head.transform.localPosition, out eastWestMask, out northSouthMask);
328
329 switch (stereoMode) {
330 case StereoscopicMode.Mono:
331 eye.stereoSeparation = 0.0f;
332 eye.transform.rotation = gameObject.transform.rotation; // Set eye's global orientation to the screen's orientation, regardless of the head's orientation.
333 eye.transform.localPosition = Vector3.zero;
334 eye.RenderToCubemap(cubemaps[(int)CubemapIndex.North], eastWestMask | northSouthMask, Camera.MonoOrStereoscopicEye.Left);
335 break;
336 case StereoscopicMode.Stereo:
337 eye.stereoSeparation = 0.0f;
338 eye.transform.rotation = gameObject.transform.rotation; // Set eye's global orientation to the screen's orientation, regardless of the head's orientation.
339 eye.transform.localPosition = new Vector3(interpupillaryDistance * -0.5f, 0.0f, 0.0f);
340 eye.RenderToCubemap(cubemaps[(int)CubemapIndex.West], eastWestMask, Camera.MonoOrStereoscopicEye.Left);
341 eye.transform.localPosition = new Vector3(interpupillaryDistance * 0.5f, 0.0f, 0.0f);
342 eye.RenderToCubemap(cubemaps[(int)CubemapIndex.East], eastWestMask, Camera.MonoOrStereoscopicEye.Right);
343 eye.transform.localPosition = new Vector3(0.0f, 0.0f, interpupillaryDistance * 0.5f);
344 eye.RenderToCubemap(cubemaps[(int)CubemapIndex.North], northSouthMask, Camera.MonoOrStereoscopicEye.Left);
345 eye.transform.localPosition = new Vector3(0.0f, 0.0f, interpupillaryDistance * -0.5f);
346 eye.RenderToCubemap(cubemaps[(int)CubemapIndex.South], northSouthMask, Camera.MonoOrStereoscopicEye.Right);
347 eye.transform.localPosition = Vector3.zero;
348 break;
349 }
350
351 material.SetInteger("_EnableStereo", stereoMode == StereoscopicMode.Stereo ? 1 : 0);
352 material.SetFloat("_CavernHeight", cavernHeight);
353 material.SetFloat("_CavernRadius", cavernRadius);
354 material.SetFloat("_CavernAngle", cavernAngle);
355 material.SetFloat("_CavernElevation", cavernElevation);
356 material.SetVector("_HeadPosition", head.transform.localPosition);
357 }
