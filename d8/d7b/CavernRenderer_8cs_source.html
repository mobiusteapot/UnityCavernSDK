141 private void RenderEyes() {
142 const int leftMask = 1 << (int)CubemapFace.PositiveX;
143 const int rightMask = 1 << (int)CubemapFace.NegativeX;
144 const int topMask = 1 << (int)CubemapFace.PositiveY;
145 const int bottomMask = 1 << (int)CubemapFace.NegativeY;
146 const int frontMask = 1 << (int)CubemapFace.PositiveZ;
147 const int backMask = 1 << (int)CubemapFace.NegativeZ;
148 const int allMask = leftMask | rightMask | topMask | bottomMask | frontMask | backMask;
149
150 // Use Camera.MonoOrStereoscopicEye.Left or Camera.MonoOrStereoscopicEye.Right to ensure that the cubemap follows the camera's rotation.
151 // Camera.MonoOrStereoscopicEye.Mono renders the cubemap to be aligned to the world's axes instead.
152 switch (stereoMode) {
153 case StereoscopicMode.Mono:
154 eye.stereoSeparation = 0.0f;
155 eye.transform.localPosition = Vector3.zero;
156 eye.transform.localRotation = Quaternion.identity;
157 eye.RenderToCubemap(cubemaps[(int)CubemapIndex.Mono], allMask | frontMask | leftMask | rightMask, Camera.MonoOrStereoscopicEye.Left);
158 break;
159 case StereoscopicMode.Stereo:
160 // eye.stereoSeparation = interpupillaryDistance;
161 eye.stereoSeparation = 0.0f;
162 eye.transform.localRotation = Quaternion.identity;
163 eye.transform.localPosition = new Vector3(-interpupillaryDistance * 0.5f, 0.0f, 0.0f);
164 eye.RenderToCubemap(cubemaps[(int)CubemapIndex.Left], allMask | frontMask, Camera.MonoOrStereoscopicEye.Left);
165 eye.transform.localPosition = new Vector3(interpupillaryDistance * 0.5f, 0.0f, 0.0f);
166 eye.RenderToCubemap(cubemaps[(int)CubemapIndex.Right], allMask | frontMask, Camera.MonoOrStereoscopicEye.Right);
167 eye.transform.localPosition = new Vector3(0.0f, 0.0f, interpupillaryDistance * 0.5f);
168 eye.RenderToCubemap(cubemaps[(int)CubemapIndex.Front], allMask | leftMask | rightMask, Camera.MonoOrStereoscopicEye.Left);
169 eye.transform.localPosition = new Vector3(0.0f, 0.0f, -interpupillaryDistance * 0.5f);
170 eye.RenderToCubemap(cubemaps[(int)CubemapIndex.Back], allMask | leftMask | rightMask, Camera.MonoOrStereoscopicEye.Right);
171 eye.transform.localPosition = Vector3.zero;
172 break;
173 }
174
175 material.SetInteger("_EnableStereo", stereoMode == StereoscopicMode.Stereo ? 1 : 0);
176
177 material.SetFloat("_CavernHeight", cavernHeight);
178 material.SetFloat("_CavernRadius", cavernRadius);
179 material.SetFloat("_CavernAngle", cavernAngle);
180 material.SetFloat("_CavernElevation", cavernElevation);
181
182 // As we learnt in computer graphics class, the inverse of a rotation matrix is also it's transpose.
183 material.SetVector("_HeadPositionInverse", -head.transform.localPosition);
184 material.SetMatrix("_HeadRotationInverse", Matrix4x4.Rotate(head.transform.localRotation).transpose);
185 }
