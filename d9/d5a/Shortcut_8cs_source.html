98 public Shortcut(DynamicMenuItemAttribute shortcutAttribute, FieldInfo field) {
99 FieldName = field.Name;
100 AllowNoneValue = shortcutAttribute.AllowNoneValue;
101
102 var constant = (string)field.GetValue(null);
103 var lastSpace = constant.LastIndexOf(' ') + 1;
104
105 if (!constant.EndsWith(" "))
106 BaseString = constant.Remove(lastSpace);
107 else {
108 BaseString = constant;
109 return;
110 }
111
112 constant = constant.Substring(lastSpace);
113
114 if (string.IsNullOrEmpty(constant))
115 return;
116
117 Ctrl = constant.Contains(CTRL_CHAR);
118 Shift = constant.Contains(SHIFT_CHAR);
119 Alt = constant.Contains(ALT_CHAR);
120
121 constant = constant.Replace(CTRL_CHAR.ToString(), string.Empty);
122 constant = constant.Replace(SHIFT_CHAR.ToString(), string.Empty);
123 constant = constant.Replace(ALT_CHAR.ToString(), string.Empty);
124 constant = constant.Replace(NONE_CHAR.ToString(), string.Empty);
125
126 KeyCode = Array.IndexOf(keys, constant);
127
128 if (KeyCode < 0 || KeyCode >= keys.Length) {
129 Logger.Warning("Invalid shortcut term: {0}", constant);
130 KeyCode = 0;
131 }
132 }

135 public string GetShortcutString() {
136 if (KeyCode == 0)
137 return "";
138
139 var result = new StringBuilder();
140
141 if (!Ctrl && !Shift && !Alt)
142 result.Append(NONE_CHAR);
143 else {
144 if (Ctrl)
145 result.Append(CTRL_CHAR);
146 if (Shift)
147 result.Append(SHIFT_CHAR);
148 if (Alt)
149 result.Append(ALT_CHAR);
150 }
151
152 result.Append(keys[KeyCode]);
153
154 return result.ToString();
155 }

246 private static void ReplaceConstant(string constantName, object newValue) {
247 try {
248 if (!IsSourceFile) {
249 Logger.Error("Could not find the source code file to change value");
250 return;
251 }
252
253 var fileText = new StringBuilder();
254 var changed = false;
255
256 using(var file = File.OpenText(ThisFilePath))
257 while (!file.EndOfStream) {
258 var line = file.ReadLine();
259
260 if (!line.Contains(constantName)) {
261 fileText.AppendLine(line);
262 continue;
263 }
264
265 var indexOfValue = line.IndexOf('=');
266
267 fileText.Append(line.Remove(indexOfValue));
268 fileText.AppendLine(string.Format("= \"{0}\";", newValue));
269 fileText.Append(file.ReadToEnd());
270
271 changed = true;
272 }
273
274 fileText = fileText.Replace("\r\n", "\n").Replace("\r", "\n").Replace("\n", Environment.NewLine);
275
276 if (changed)
277 File.WriteAllText(ThisFilePath, fileText.ToString());
278 else
279 Logger.Warning("Failed to find field {0} on {1}", constantName, ThisFilePath);
280 } catch (Exception e) {
281 Logger.Exception(e);
282 Logger.Error("Failed to save Fullscreen Editor shortcuts");
283 }
