84
85 var openFullscreens = Fullscreen.GetAllFullscreen();
86
87 if (openFullscreens.Length > 0) {
88 foreach (var fs in openFullscreens)
89 fs.Close();
90 return;
91 }
92
93 var displays = DisplayInfo
94 .GetDisplays()
95 .Where(d => (d.displayDevice.StateFlags & DisplayDeviceStateFlags.AttachedToDesktop) != 0)
96 .ToList();
97
98 for (var i = 0; i < displays.Count && i < 8; i++) {
99 var targetDisplay = FullscreenPreferences.MosaicMapping.Value[i];
100
101 if (targetDisplay < 0) {
102 continue; // -1 means none
103 }
104
105 var candidate = FindCandidateForFullscreen(Types.GameView, FullscreenUtility.GetMainGameView());
106
107 if (candidate) {
108 candidate = EditorWindow.Instantiate(candidate);
109 candidate.Show();
110 }
111
112 var fs = ScriptableObject.CreateInstance<FullscreenWindow>();
113 var rect = displays[i].UnityCorrectedArea;
114 fs.OpenWindow(rect, Types.GameView, candidate, true);
115
116 var gameView = fs.ActualViewPyramid.Window;
117
118 FullscreenUtility.SetGameViewDisplayTarget(gameView, targetDisplay);

120 }

155 private static EditorWindow FindCandidateForFullscreen(Type type, EditorWindow mainCandidate = null) {
156 if (type == null)
157 throw new ArgumentNullException("type");
158
159 if (!type.IsOfType(typeof(EditorWindow)))
160 throw new ArgumentException("Invalid type, type must inherit from UnityEditor.EditorWindow", "type");
161
162 if (mainCandidate && !mainCandidate.IsOfType(type))
163 throw new ArgumentException("Main candidate type must match the type argument or be null", "mainCandidate");
164
165 // if (mainCandidate && !Fullscreen.GetFullscreenFromView(mainCandidate))
166 if (mainCandidate)
167 return mainCandidate; // Our candidate is not null and is not fullscreened either
168
169 return Resources // Returns the first window of our type that is not in fullscreen
170 .FindObjectsOfTypeAll(type)
171 .Cast<EditorWindow>()
172 .FirstOrDefault(window => !Fullscreen.GetFullscreenFromView(window));
173 }
