43 private static void SetIsPlaying(bool playing) {
44
45 var fullscreens = Fullscreen.GetAllFullscreen()
46 .Select(fullscreen => fullscreen as FullscreenWindow)
47 .Where(fullscreen => fullscreen);
48
49 // We close all the game views created on play, even if the option was disabled in the middle of the play mode
50 // This is done to best reproduce the default behaviour of the maximize on play
51 if(!playing) {
52 foreach(var fs in fullscreens)
53 if(fs && fs.CreatedByFullscreenOnPlay) // fs might have been destroyed
54 fs.Close();
55 return;
56 }
57
58 if(!FullscreenPreferences.FullscreenOnPlayEnabled)
59 return; // Nothing to do here
60
61 if(FullscreenUtility
62 .GetGameViews()
63 .Any(gv => {
64 if(!gv) return false;
65
66 return PlaymodeBehaviourImplemented(gv) && GetEnterPlayModeBehavior(gv) == CustomEnterPlayModeBehavior.PlayFullscreen;
67 })) {
68 EditorUtility.DisplayDialog("Fullscreen on play conflict", "Seems like you have both Unity's built-in fullscreen on play and Fullscreen Editor's plugin enabled. Please, make sure you only have one of them enabled to prevent conflicts.", "Got it!");
69 FullscreenPreferences.FullscreenOnPlayEnabled.Value = false;
70 return;
71 }
72
73 var gameView = FullscreenOnPlayGameView();
74
75 if(!gameView) // no gameview has the fullscreen on play option enabled
76 return;
77
78 foreach(var fs in fullscreens)
79 if(fs && fs.Rect.Overlaps(gameView.position)) // fs might have been destroyed
80 return; // We have an open fullscreen where the new one would be, so let it there
81
82 if(gameView && Fullscreen.GetFullscreenFromView(gameView))
83 return; // The gameview is already in fullscreen
84
85 var gvfs = Fullscreen.MakeFullscreen(Types.GameView, gameView);
