35 public static void BringWindowsAbove() {
36
37 if (!FullscreenPreferences.KeepFullscreenBelow)
38 return;
39
40 var fullscreens = Fullscreen.GetAllFullscreen();
41 if (fullscreens.Length == 0)
42 return;
43
44 var methodName = "Internal_BringLiveAfterCreation";
45 var windows = GetAllContainerWindowsOrdered()
46 .Where(w => !Fullscreen.GetFullscreenFromView(w))
47 .Where(w => {
48 if (w.GetPropertyValue<int>("showMode") == (int)ShowMode.MainWindow)
49 return false; // Main Window should be kept below everything
50
51 if (fullscreens.FirstOrDefault((f) => f.m_src.Container == w))
52 return false; // Keep other fullscreen containers below
53
54 return true;
55 });
56
57 foreach (var w in windows) {
58 if (w.HasMethod(methodName, new Type[] { typeof(bool), typeof(bool), typeof(bool) }))
