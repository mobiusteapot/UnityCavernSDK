58 private void Update() {
59 if (EditorApplication.timeSinceStartup < m_nextUpdate)
60 return;
61
62 m_nextUpdate = EditorApplication.timeSinceStartup + (1f / PREVIEW_FRAMERATE);
63 RenderTexture.ReleaseTemporary(m_previewRT);
64 m_previewRT = null;
65
66 if (!PreviewSupported || m_containerForcefullyClosed)
67 return;
68
69 var view = FullscreenContainer ? FullscreenContainer.m_dst.View : null;
70 var width = (int)FullscreenContainer.Rect.width;
71 var height = (int)FullscreenContainer.Rect.height;
72
73 if (!view || width < 10 || height < 10)
74 return;
75
76 m_previewRT = RenderTexture.GetTemporary(width, height, 0);
77 view.InvokeMethod("GrabPixels", m_previewRT, new Rect(0f, 0f, width, height));
