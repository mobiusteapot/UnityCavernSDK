122 /// <summary>Open a new fullscreen if there's none open, otherwise, close the one already open.</summary>
123 /// <param name="window">The window that will go fullscreen. If null a new one will be instantiated based on the given type.</param>
124 /// <param name="type">The type of the window to instantiate if the given window is null.</param>
125 public static void ToggleFullscreen(Type type, EditorWindow window = null) {
126 var rect = FullscreenRects.GetFullscreenRect(FullscreenPreferences.RectSource, window);
127 var oldFullscreen = GetFullscreenFromView(window);
128
129 if(oldFullscreen) {
130 oldFullscreen.Close();
131 return;
132 }
133
134 oldFullscreen = GetFullscreenOnRect(rect);
135
136 var newFullscreen = MakeFullscreen(type, window);
137
138 newFullscreen.didPresent += () => {
139 if(oldFullscreen)
140 oldFullscreen.Close();
141 };
142 }
