174 /// <summary>Does the given <see cref="MenuItem"/> path contains a key binding?</summary>
175 public static bool MenuItemHasShortcut(string menuItemPath) {
176 var index = menuItemPath.LastIndexOf(" ");
177
178 if(index++ == -1)
179 return false;
180
181 var shortcut = menuItemPath.Substring(index).Replace("_", "");
182 var evt = Event.KeyboardEvent(shortcut);
183
184 shortcut = InternalEditorUtility.TextifyEvent(evt);
185
186 return !shortcut.Equals("None", StringComparison.InvariantCultureIgnoreCase);
187 }

273 /// <summary>Returns the focused view if it is a dock area with more than one tab, otherwise, returns the focused window.</summary>
274 public static ScriptableObject GetFocusedViewOrWindow() {
275 var mostSpecificView = Types.GUIView.GetPropertyValue<ScriptableObject>("focusedView");
276
277 if(!mostSpecificView)
278 return null;
279
280 // The most specific obj is a window, open it instead of the view
281 if(mostSpecificView.IsOfType(Types.HostView, false))
282 return EditorWindow.focusedWindow;
283
284 var viewHierarchy = GetViewHierarchy(mostSpecificView);
285 var leastSpecificView = viewHierarchy.LastOrDefault();
286
287 // The view hierarchy has the same length of all the views in this ContainerWindow
288 // This means there are no cousins views handled by a split group on the surroundings of this one
289 // So, we're alone on the container
290 var alone = leastSpecificView.GetPropertyValue<Array>("allChildren").Length == viewHierarchy.Length;
291
292 if(alone && EditorWindow.focusedWindow.InvokeMethod<int>("GetNumTabs") > 1)
293 alone = false; // But, we may not be the only tab on the host view
294
295 // If the focused view is in the main view, or we are the only child on this view, then we open the window
296 return alone || leastSpecificView.IsOfType(Types.MainView) ?
297 EditorWindow.focusedWindow :
298 leastSpecificView;
299 }

301 /// <summary>Returns all the parents of a given view.</summary>
302 public static ScriptableObject[] GetViewHierarchy(ScriptableObject view) {
303 if(!view)
304 return new ScriptableObject[0];
305
306 view.EnsureOfType(Types.View);
307
308 var list = new List<ScriptableObject>() { view };
309 var parent = view.GetPropertyValue<ScriptableObject>("parent");
310
311 while(parent) { // Get the least specific view
312 view = parent;
313 list.Add(view);
314 parent = view.GetPropertyValue<ScriptableObject>("parent");
315 }
316
317 return list.ToArray();
318 }

345 /// <summary>Focus a view.</summary>
346 public static void FocusView(ScriptableObject guiView) {
347 if(!guiView)
348 return;
349
350 // guiView.EnsureOfType(Types.GUIView);
351 if(guiView.IsOfType(Types.GUIView))
352 guiView.InvokeMethod("Focus");
353 else {
354 var vp = new ViewPyramid(guiView);
355 var vc = vp.Container;
356 var methodName = "Internal_BringLiveAfterCreation";
357
358 if(vc) {
359 if(vc.HasMethod(methodName, new Type[] { typeof(bool), typeof(bool), typeof(bool) }))
360 // displayImmediately, setFocus, showMaximized
361 vc.InvokeMethod(methodName, false, true, false);
362 else
363 // displayImmediately, setFocus
364 vc.InvokeMethod(methodName, false, true);
365 }
